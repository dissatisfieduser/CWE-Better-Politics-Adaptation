ideology_maoist = {
	icon = "gfx/interface/icons/ideology_icons/ideology_leader/maoist.dds"
	
	character_ideology = yes
	
	possible = {
		scope:interest_group = { bpm_ig_enabled = yes }
		# Agitator ideologies should disapprove of a current law
		NAND = {
			has_role = agitator
			owner = {
				has_law = law_type:law_council_republic
				AND = {
					bpm_country_laws_has_socialist_economics = yes
					NOT = { has_law = law_type:law_cooperative_ownership }
				}
				bpm_country_laws_are_very_repressive = yes
				has_law = law_type:law_socialist_labor_unions
			}
		}
		# State forces outside of CR should not be communist
		NAND = {
			owner = { NOT = { has_law = law_type:law_council_republic } }
			OR = {
				is_ruler = yes
				is_heir = yes
				scope:interest_group = { bpm_ig_is_state_institutional = yes }
			}
		}
		scope:interest_group = {
			OR = {
				bpm_ig_is_revolutionary_socialist = yes
				bpm_ig_is_middle_institutional = yes
				bpm_ig_is_rural_institutional = yes
			}
			OR = {
				bpm_ig_is_socialist = yes
				AND = {
					owner = {
						any_interest_group = {
							bpm_ig_is_socialist = yes
							ig_clout > 0.1
						}
					}
					has_global_variable = second_intl_chain_complete_var
				}			}
		}
		# Monarchs and religious heads cannot be vanguardists
		NAND = {
			owner = {
				OR = {
					has_law = law_type:law_monarchy
					has_law = law_type:law_theocracy
				}
			}
			OR = {
				is_ruler = yes
				is_heir = yes
			}
		}
	}
	
	lawgroup_governance_principles = {
		law_monarchy = strongly_disapprove
		law_presidential_republic = disapprove
		law_theocracy = disapprove
		law_parliamentary_republic = neutral
		law_council_republic = strongly_approve
	}

	lawgroup_land_reform = {
		law_serfdom = strongly_disapprove
		law_tenant_farmers = disapprove
		law_homesteading = disapprove
		law_collectivized_agriculture = strongly_approve
		law_commercialized_agriculture = disapprove
	}
	
	lawgroup_free_speech = {
		law_protected_speech = strongly_disapprove
		law_right_of_assembly = disapprove
		law_censorship = approve
		law_outlawed_dissent = strongly_approve
	}
	lawgroup_economic_system = {	
		law_command_economy = strongly_approve
		law_interventionism = neutral
		law_agrarianism = disapprove
		law_traditionalism = strongly_disapprove		
		law_laissez_faire = disapprove		
		law_cooperative_ownership = neutral
		law_industry_banned = strongly_disapprove
		law_semi_command_economy = approve
		law_state_capitalism = disapprove
		law_social_capitalism = disapprove
	}			
	lawgroup_labor_unions = {
		law_illegal_labor_unions = strongly_disapprove
		law_state_controlled_labor_unions = strongly_disapprove
		law_limited_labor_unions = strongly_disapprove
		law_all_labor_unions_allowed = neutral
		law_socialist_labor_unions = strongly_approve
		law_syndicalism = neutral
	}
	lawgroup_foreign_policy = {
		law_ultramilitarism = approve
		law_jingoism = strongly_approve
		law_flexible_foreign_policy = neutral
		law_justum_bellum = strongly_disapprove
		law_dove_politics = strongly_disapprove
		law_armed_neutrality = strongly_disapprove
		law_disarmed_nation = strongly_disapprove
		law_premodern_diplomacy = strongly_disapprove
	}
	lawgroup_slavery = {
		law_slavery_banned = disapprove
		law_slavery_political = strongly_approve
		law_slavery_cultural = disapprove
	}
	lawgroup_army_model = {
		law_peasant_levies = strongly_disapprove
		law_national_militia = strongly_approve
		law_professional_army = disapprove
		law_mass_conscription = approve
	}

	leader_weight = {
		value = -10
		# More likely for revolutionary socialists
		if = {
			limit = {
				scope:interest_group = {
					bpm_ig_is_revolutionary_socialist = yes
				}
			}
			add = 100
		}
		# More likely when SoL is low outside of council republic
		if = {
			limit = {
				scope:interest_group = {
					is_interest_group_type = ig_revolutionist_socialists
				}
				owner = {
					NOT = { has_law = law_type:law_council_republic }
					average_sol <= 10
				}
			}
			add = 50
		}
		# More likely in an autocracy
		if = {
			limit = {
				scope:interest_group = {
					is_interest_group_type = ig_revolutionist_socialists
				}
				owner = {
					OR = {
						has_law = law_type:law_no_voting
						has_law = law_type:law_rigged_voting
					}
				}
			}
			add = 50
		}
		# Unlikely for content IG's under anarchism
		if = {
			limit = {
				owner = {
					has_law = law_type:law_anarchy
				}
				scope:interest_group = {
					ig_approval >= 5
				}
			}
			add = -100
		}
		# Unlikely for Armed Forces outside of Council Republic
		if = {
			limit = {
				owner = {
					NOT = { has_law = law_type:law_council_republic }
				}
				scope:interest_group = {
					is_interest_group_type = ig_armed_forces
				}
			}
			add = -100
		}
		# Impossible for Bureaucrats outside of Council Republic
		if = {
			limit = {
				owner = {
					NOT = { has_law = law_type:law_council_republic }
				}
				scope:interest_group = {
					is_interest_group_type = ig_bureaucrats
				}
			}
			add = -1000
		}
		
		# vanguardists generally don't come from reactionary igs
		if = {
			limit = {
				scope:interest_group = {
					OR = {
						is_interest_group_type = ig_conservatives
						is_interest_group_type = ig_landowners
						is_interest_group_type = ig_devout
						is_interest_group_type = ig_industrialists
						is_interest_group_type = ig_liberals
						is_interest_group_type = ig_reactionaries
						is_interest_group_type = ig_national_liberals
						is_interest_group_type = ig_market_liberals
					}
				}
			}
			add = -750
		}
		
		# vanguardists generally come out of revolutionary igs
		if = {
			limit = {
				scope:interest_group = {
					OR = {
						is_interest_group_type = ig_rural_folk
						is_interest_group_type = ig_socialists
						is_interest_group_type = ig_agrarian_populists
						is_interest_group_type = ig_trade_unions
					}
				}
			}
			add = 100
		}
		
		# IGs in power are more likely to have autocratic opinions
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			add = 50
		}
		
		# more likely if you border a rival
		if = {
			limit = {
				owner = {
					any_rival_country = {
						any_neighbouring_state = {
							owner = root.owner
						}
					}
				}
			}
			add = 25
		}
		# more likely if poor
		if = {
			limit = {
				standard_of_living < 10
			}
			add = 200
		}
		
		# Cycle of life
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = no
				}
				owner = { is_a_democracy = yes }
			}
			add = 50
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = no
				}
				owner = { is_a_dictatorship = yes }
			}
			add = -50
		}
		
		# National Values
		if = {
			limit = {
				owner = { has_modifier = equality_national_value }
			}
			add = 25
		}
		
		# World Tension
		if = {
			limit = {
				owner = { has_global_variable = very_high_world_tension }
			}
			add = 25
		}
		
	}
}
